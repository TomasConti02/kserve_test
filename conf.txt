helm repo add kiali https://kiali.org/helm-charts
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
helm repo add grafana https://grafana.github.io/helm-charts
helm repo update



# Prometheus (il database delle metriche)
helm install prometheus prometheus-community/prometheus \
  --namespace istio-system \
  --set server.persistentVolume.enabled=false \
  --set alertmanager.enabled=false \
  --set pushgateway.enabled=false

# Grafana (la dashboard)
helm install grafana grafana/grafana \
  --namespace istio-system \
  --set persistence.enabled=false \
  --set adminPassword="admin"





# Collega Kiali a Prometheus
kubectl patch kiali kiali -n istio-system --type merge \
  -p '{"spec":{"external_services":{"prometheus":{"url":"http://prometheus-server.istio-system:80"}}}}'

# Collega Kiali a Grafana
kubectl patch kiali kiali -n istio-system --type merge \
  -p '{"spec":{"external_services":{"grafana":{"url":"http://grafana.istio-system:80"}}}}'

# Riavvio necessario per applicare le modifiche
kubectl rollout restart deployment kiali -n istio-system



kubectl port-forward svc/kiali -n istio-system 20001:20001
# Apri: http://localhost:20001



kubectl port-forward svc/grafana -n istio-system 3000:80
# Apri: http://localhost:3000 (User: admin, Pass: admin)
